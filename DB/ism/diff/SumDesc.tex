A pillanatnyi változási gyorsaság,
az érintő $x$ tengellyel bezárt szögének tangense, meredekség.
\Mat{
   f'(x_0)=\lim_{h\to 0} \frac{f(x_0+h)-f(x_0)}{h}
}
Deriválás és műveletek (legyen $C\in\mathbb{R}$):
\Mat{
   (Cf)'=cf'\hspace{1cm}\\
   (f+g)=f'+g'\\
   (fg)'=f'g+fg'\\
   \left( \frac{f}{g}\right) '=\frac{f'g-fg'}{g^2}\hspace{1cm} g\neq 0\\
   f(g)'=f'(g)g'\\
   (f^{-1})'(x)=\frac{1}{f'(f^{-1}(x))}
}
\Dnew
Elemi \fv{ek:}
\Mat{
   C' =  0\\
   (x^{C})' = Cx^{C-1}\\
   \sin'=\cos\\
   \cos'=-\sin\\
   \tg'=\frac{1}{\cos^2}\\
   \ctg'=-\frac{1}{\sin^2}\\
   (C^x)'=\log(C)C^x\hspace{1cm}C>0\\
   \log(|x|)'=\frac{1}{x}\hspace{1cm}x\neq0
}
